交互式 CSS 和 JS

使用以下代码来创建一个交互式的 CSS 和 JS 示例：


				// CSS
				.interactive {
					background-color: #f0f0f0;
					padding: 20px;
					border: 1px solid #ddd;
					border-radius: 10px;
					box-shadow: 0 0 10px rgba(0, 0, 0, 0.1);
				}
				
				.interactive code {
					background-color: #f0f0f0;
					padding: 10px;
					border: 1px solid #ddd;
					border-radius: 5px;
					font-size: 16px;
					font-family: monospace;
				}
				
				// JS
				document.addEventListener("DOMContentLoaded", function() {
					var interactive = document.querySelector(".interactive");
					interactive.addEventListener("mouseover", function() {
						interactive.style.background = "#ccc";
					});
					interactive.addEventListener("mouseout", function() {
						interactive.style.background = "#f0f0f0";
					});
				});
